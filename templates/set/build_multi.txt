  blocks, ok := BuildSet[models.{MODEL}]({OPERATOR_INSTANCE_NAME}.Ctx, {OPERATOR_INSTANCE_NAME}.Diags, state)

  // Return nil if conversion failed or set was empty
  if !ok {
    return nil
  }

  // Prepare the output slice
  output := []client.{CLIENT_TYPE}{}

  // Iterate over each block and construct an output item
  for _, block := range blocks {

}
